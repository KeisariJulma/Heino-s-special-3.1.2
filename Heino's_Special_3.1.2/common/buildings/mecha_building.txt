building_titan_forge = {
	base_buildtime = 480
	planet_unique = yes
	empire_unique = yes
	base_cap_amount = 1

	category = army
	
	potential = {
		owner = { has_ascension_perk = ap_the_flesh_is_weak }
		planet = {
			NOT = { is_planet_class = pc_habitat }	
		}
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	army_modifier = {
		army_starting_experience_add = 100
	}
	
	triggered_planet_modifier = {
		potential = {
			owner = { is_gestalt = no }
		}
		modifier = {
			job_roboticist_add = 4
			job_soldier_add = 4
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 1000
		}			
		upkeep = {
			energy = 10
		}
	}
	
	prerequisites = {
		"tech_mega_engineering"
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
	
	ai_weight = {
		weight = 0.5
	}
}
