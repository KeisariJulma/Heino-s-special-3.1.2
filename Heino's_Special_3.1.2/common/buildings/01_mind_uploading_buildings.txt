# Regular empire
building_virtual_reality_center = {
	base_buildtime = 360

	potential = {
		exists = owner
		NOT = { has_modifier = slave_colony }
		owner = {
			has_policy_flag = brain_uploading_allowed
			NOR = {
				is_machine_empire = yes
				is_spiritualist = yes
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			has_modifier = slave_colony
			owner = {
				OR = {
					is_gestalt = yes
					is_spiritualist = yes
				}
			}
		}
	}

	convert_to = {
		building_virtual_reality_gestalt_center
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 10
			consumer_goods = 2
		}
	}

	prerequisites = {
		"virtual_reality_building"
	}
	
	planet_modifier = {
		job_sysadmin_add = 2
	}

	show_tech_unlock_if = {
		is_gestalt = no
		is_spiritualist = no
	}

	triggered_desc = {
		text = job_sysadmin_desc
	}

	ai_weight = {
		weight = 0
	}
}
