district_trp_housing = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = refinery_world
	}

	potential = {
		uses_district_set = refinery_world
	}

	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 5
		}
	}

	planet_modifier = {
		planet_housing_add = 15
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 9
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_public_works }
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				NOT = { has_valid_civic = civic_agrarian_idyll }
			}
		}
		modifier = {
			planet_housing_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
				has_ethic = ethic_gestalt_consciousness
			}
		}
		modifier = {
			job_maintenance_drone_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		modifier = {
			job_clerk_add = 4
			job_merchant_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_non_swapped_tradition = tr_prosperity_interstellar_franchising }
		}
		modifier = {
			job_clerk_add = 4
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		
		text = job_clerk_effect_desc
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
			}
		}
		
		text = job_merchant_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			free_housing < 10
		}
	}
}
